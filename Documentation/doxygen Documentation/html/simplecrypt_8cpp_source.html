58 void SimpleCrypt::splitKey()
59 {
60  m_keyParts.clear();
61  m_keyParts.resize(8);
62  for (int i = 0; i < 8; i++)
63  {
64  quint64 part = m_key;
65  for (int j = i; j > 0; j--)
66  part = part >> 8;
67  part = part & 0xff;
68  m_keyParts[i] = static_cast<char>(part);
69  }
70 }

78 QByteArray SimpleCrypt::encryptToByteArray(QByteArray plaintext)
79 {
80  if (m_keyParts.isEmpty())
81  {
82  qWarning() << "No key set.";
83  m_lastError = ErrorNoKeySet;
84  return QByteArray();
85  }
86 
87  QByteArray ba = plaintext;
88 
89  CryptoFlags flags = CryptoFlagNone;
90  if (m_compressionMode == CompressionAlways)
91  {
92  ba = qCompress(ba, 9); //maximum compression
93  flags |= CryptoFlagCompression;
94  }
95  else if (m_compressionMode == CompressionAuto)
96  {
97  QByteArray compressed = qCompress(ba, 9);
98  if (compressed.count() < ba.count())
99  {
100  ba = compressed;
101  flags |= CryptoFlagCompression;
102  }
103  }
104 
105  QByteArray integrityProtection;
106  if (m_protectionMode == ProtectionChecksum)
107  {
108  flags |= CryptoFlagChecksum;
109  QDataStream s(&integrityProtection, QIODevice::WriteOnly);
110  s << qChecksum(ba.constData(), ba.length());
111  }
112  else if (m_protectionMode == ProtectionHash)
113  {
114  flags |= CryptoFlagHash;
115  QCryptographicHash hash(QCryptographicHash::Sha1);
116  hash.addData(ba);
117 
118  integrityProtection += hash.result();
119  }
120 
121  //prepend a random char to the string
122  char randomChar = char(qrand() & 0xFF);
123  ba = randomChar + integrityProtection + ba;
124 
125  int pos(0);
126  char lastChar(0);
127 
128  int cnt = ba.count();
129 
130  while (pos < cnt)
131  {
132  ba[pos] = ba.at(pos) ^ m_keyParts.at(pos % 8) ^ lastChar;
133  lastChar = ba.at(pos);
134  ++pos;
135  }
136 
137  QByteArray resultArray;
138  resultArray.append(char(0x03)); //version for future updates to algorithm
139  resultArray.append(char(flags)); //encryption flags
140  resultArray.append(ba);
141 
142  m_lastError = ErrorNoError;
143  return resultArray;
144 }

186 QByteArray SimpleCrypt::decryptToByteArray(QByteArray cypher)
187 {
188  if (m_keyParts.isEmpty())
189  {
190  qWarning() << "No key set.";
191  m_lastError = ErrorNoKeySet;
192  return QByteArray();
193  }
194 
195  QByteArray ba = cypher;
196 
197  if (cypher.count() < 3)
198  return QByteArray();
199 
200  char version = ba.at(0);
201 
202  if (version != 3)
203  { //we only work with version 3
204  m_lastError = ErrorUnknownVersion;
205  qWarning() << "Invalid version or not a cyphertext.";
206  return QByteArray();
207  }
208 
209  CryptoFlags flags = CryptoFlags(ba.at(1));
210 
211  ba = ba.mid(2);
212  int pos(0);
213  int cnt(ba.count());
214  char lastChar = 0;
215 
216  while (pos < cnt)
217  {
218  char currentChar = ba[pos];
219  ba[pos] = ba.at(pos) ^ lastChar ^ m_keyParts.at(pos % 8);
220  lastChar = currentChar;
221  ++pos;
222  }
223 
224  ba = ba.mid(1); //chop off the random number at the start
225 
226  bool integrityOk(true);
227  if (flags.testFlag(CryptoFlagChecksum))
228  {
229  if (ba.length() < 2)
230  {
231  m_lastError = ErrorIntegrityFailed;
232  return QByteArray();
233  }
234  quint16 storedChecksum;
235  {
236  QDataStream s(&ba, QIODevice::ReadOnly);
237  s >> storedChecksum;
238  }
239  ba = ba.mid(2);
240  quint16 checksum = qChecksum(ba.constData(), ba.length());
241  integrityOk = (checksum == storedChecksum);
242  }
243  else if (flags.testFlag(CryptoFlagHash))
244  {
245  if (ba.length() < 20)
246  {
247  m_lastError = ErrorIntegrityFailed;
248  return QByteArray();
249  }
250  QByteArray storedHash = ba.left(20);
251  ba = ba.mid(20);
252  QCryptographicHash hash(QCryptographicHash::Sha1);
253  hash.addData(ba);
254  integrityOk = (hash.result() == storedHash);
255  }
256 
257  if (!integrityOk)
258  {
259  m_lastError = ErrorIntegrityFailed;
260  return QByteArray();
261  }
262 
263  if (flags.testFlag(CryptoFlagCompression))
264  ba = qUncompress(ba);
265 
266  m_lastError = ErrorNoError;
267  return ba;
268 }
